// ACE SOLVE 根据同济滤波器修改,针对英雄打前哨站所修改
    auto_aim_interfaces::msg::Gimbal CoordSolver::new_solve_ace_outpost(const auto_aim_interfaces::msg::Target &target,
                                                                const std::unique_ptr<rm_auto_aim::Tracker> &tracker,
                                                                const rclcpp::Time &current_time,
                                                                std::shared_ptr<tf2_ros::Buffer> tf2_buffer_)
    {
        // Get current roll, yaw and pitch of gimbal
        // auto start = std::chrono::high_resolution_clock::now();
        try
        {
            auto gimbal_tf = tf2_buffer_->lookupTransform("odom", "gimbal_link", tf2::TimePointZero);
            auto msg_q = gimbal_tf.transform.rotation;

            tf2::Quaternion tf_q;
            tf2::fromMsg(msg_q, tf_q);
            tf2::Matrix3x3(tf_q).getRPY(rpy_[0], rpy_[1], rpy_[2]);
            rpy_[1] = -rpy_[1];
            // std::cout << "current roll = " << rpy_[0] << std::endl;
            // std::cout << "current pitch = " << rpy_[1] << std::endl;
            // std::cout << "current yaw = " << rpy_[2] << std::endl;
        }
        catch (tf2::TransformException &ex)
        {
            RCLCPP_ERROR(rclcpp::get_logger("armor_tracker"), "coordsolver:%s", ex.what());
            throw ex;
        }
        //关于dz的先验值 102mm 102mm 204mm 三个板高度分别为 1114 1216 1318
        auto_aim_interfaces::msg::Gimbal gimbal_cmd;
        //计算当前观测到的板的高度，算等待板
        int now_id = tracker->now_id_;
        double r = 0.275;
        double x_c = tracker->target_state(0);
        double y_c = tracker->target_state(2);
        double v_yaw;
        if (target.v_yaw > 1.0f)
            v_yaw = 2.513274123; // M_PI*0.8  3.14159265358979323846 * 0.8
        else if (target.v_yaw < -1.0f)
            v_yaw = -2.513274123;
        else
            v_yaw = 0;
        Eigen::Vector4d target_vec(0.0 ,0.0 ,0.0 ,v_yaw);
        Eigen::Vector4d WaitPosition;
        Eigen::Vector4d targetPosition;
        // double x_a = x_c - r * std::cos(rpy_[2]);
        // double y_a = y_c - r * std::sin(rpy_[2]);
        double x_a = x_c - r * std::cos(tracker->target_state(6));
        double y_a = y_c - r * std::sin(tracker->target_state(6));
        //计算打等待板飞行时间
        WaitPosition = {x_a,y_a,tracker->target_state(4),tracker->target_state(6)};
        //用飞行时间去预测，然后再算等待板，再算飞行时间，知道当前飞行时间与上次飞行时间差0.001的时候，视为收敛
        double now_flying_time,last_flying_time;
        double yaw,pitch;
        yaw = atan2(WaitPosition(1),WaitPosition(0));
        PESolver pesolver(this->resistance,this->gravity,this->velocity);
        pitch = pesolver.solvepitch(sqrtf(WaitPosition(0)*WaitPosition(0)+WaitPosition(1)*WaitPosition(1)),WaitPosition(2));
        now_flying_time = getFlyingTime(WaitPosition,pitch);
        last_flying_time = now_flying_time;
        double dt;
        int idx = 0;
        bool converged = false;
        for(int iter = 0;iter < 10;++iter)
        {
            if(v_yaw < 0.001)
            {
                converged = true;
                break;
            }
            dt = (current_time - rclcpp::Time(target.header.stamp)).seconds() + now_flying_time + bias_time;
            targetPosition = WaitPosition + target_vec * dt;
            Eigen::Matrix<double, 11, 1> current_state =  Eigen::Matrix<double, 11, 1>{x_c, 0., y_c, 0., targetPosition(2), 0., targetPosition(3),
                                                0., tracker->target_state(8), tracker->target_state(9), tracker->target_state(10)};
            std::vector<Eigen::Vector4d> armor_position = getAllArmorPositions(tracker, current_state);
            std::pair<bool, int> result = chooseBestIdx(tracker, armor_position);
            idx = result.second;
            // armor_position[idx](0) = x_c - r * std::cos(armor_position[idx](3));
            // armor_position[idx](1) = x_c - r * std::cos(armor_position[idx](3));
            armor_position[idx](0) = x_c - r * std::cos(rpy_[2]);
            armor_position[idx](1) = x_c - r * std::cos(rpy_[2]);
            // armor_position[idx](3) = rpy_[2];
            targetPosition = armor_position[idx];
            pitch = pesolver.solvepitch(sqrtf(targetPosition(0)*targetPosition(0)+targetPosition(1)*targetPosition(1)),targetPosition(2));
            now_flying_time = getFlyingTime(targetPosition,pitch);
            if(std::abs(now_flying_time - last_flying_time) < 0.001)
            {
                std::cout<<"iter:"<<iter<<"  -------------outpost flying time converged----------------"<<std::endl;
                converged = true;
                break;
            }
            last_flying_time = now_flying_time;
        }

        gimbal_cmd.header = target.header;
        if(converged)
        {
            //火控
            if(v_yaw > 0.01)
            {
                aim_armor_idx = idx;
                double alpha = rpy_[2];
                // double beta = target_aim(3);  
                double beta = targetPosition(3);
                // RCLCPP_INFO(get_logger(),"yaw:%f,target_aim:%f",alpha,beta);
                // std::cout<<"alpha:"<<alpha<<"pitch:"<<pitch<<std::endl;
                Eigen::Matrix2d R_odom2center,R_odom2armor;
                R_odom2center << std::cos(alpha), std::sin(alpha),
                    -std::sin(alpha), std::cos(alpha);
                R_odom2armor << std::cos(beta), std::sin(beta),
                    -std::sin(beta), std::cos(beta);

                Eigen::Matrix2d R_center2armor = R_odom2center.transpose() * R_odom2armor;
                double decision_angle = -std::asin(R_center2armor(0, 1));
                gimbal_cmd.fire_advice = std::fabs(decision_angle) * 180.0 / M_PI <= 9.0 ? true : false;
            }
            else
            {
                gimbal_cmd.fire_advice = false;
            }

            gimbal_cmd.header = target.header;
            gimbal_cmd.pitch = pitch;
            gimbal_cmd.yaw = yaw * 180 / M_PI; //把设定的yaw轴转换到imu设定值,把弧度转化为度数
            gimbal_cmd.pitch += pitch_compensate;
            gimbal_cmd.yaw += yaw_compensate;
            gimbal_cmd.position.x = targetPosition(0);
            gimbal_cmd.position.y = targetPosition(1);
            gimbal_cmd.position.z = targetPosition(2);
            gimbal_cmd.imu_pitch = rpy_[1] * 180 / M_PI;
            gimbal_cmd.imu_yaw = rpy_[2] *  180 / M_PI;
        }
        else
        {
            gimbal_cmd.distance = -1;
        }

        return gimbal_cmd;
    }
